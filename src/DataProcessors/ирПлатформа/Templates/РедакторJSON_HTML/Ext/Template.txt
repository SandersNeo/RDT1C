  function Init() {

  const container = document.getElementById("jsoneditor");
  let lastSelectedPath = '';
  let staticNavAdded = false;
  let copyButtonAdded = false;

  function showTooltip(element, message, isError = false) {
    const rect = element.getBoundingClientRect();
    const tooltip = document.createElement('div');
    tooltip.className = `copy-tooltip ${isError ? 'error' : 'success'}`;
    tooltip.textContent = message;
    tooltip.style.left = (rect.right - 20) + 'px';
    tooltip.style.top = (rect.bottom + 8) + 'px';
    tooltip.style.transform = 'translateX(-100%)';
    document.body.appendChild(tooltip);
    setTimeout(() => {
      if (tooltip.parentNode) tooltip.remove();
    }, 1500);
  }

  function getCurrentPath() {
    const pathDisplay = document.querySelector('.jsoneditor-treepath');
    if (!pathDisplay) return '';
    const elements = pathDisplay.querySelectorAll('.jsoneditor-treepath-element');
    if (!elements.length) return '';
    const parts = [];
    for (let i = 1; i < elements.length; i++) {
      let part = elements[i].textContent.trim();
      if (part.startsWith('"') && part.endsWith('"')) part = part.slice(1, -1);
      parts.push(part);
    }
    return parts.join('.');
  }

  function selectNodeByPath(pathStr, sourceElement = null) {
    if (!pathStr) {
      if (sourceElement) showTooltip(sourceElement, 'Path is empty', true);
      return false;
    }

    // Проверяем существование пути в данных
    try {
      const data = editor.get();
      const parts = pathStr.split('.');
      let current = data;
      for (let i = 0; i < parts.length; i++) {
        const part = parts[i];
        if (current === null || current === undefined) {
          if (sourceElement) showTooltip(sourceElement, 'Path not found: ' + pathStr, true);
          return false;
        }
        if (Array.isArray(current)) {
          const idx = parseInt(part, 10);
          if (isNaN(idx)) {
            if (sourceElement) showTooltip(sourceElement, 'Invalid array index: ' + part, true);
            return false;
          }
          current = current[idx];
        } else if (typeof current === 'object') {
          if (!current.hasOwnProperty(part)) {
            if (sourceElement) showTooltip(sourceElement, 'Property not found: ' + part, true);
            return false;
          }
          current = current[part];
        } else {
          if (sourceElement) showTooltip(sourceElement, 'Cannot navigate further', true);
          return false;
        }
      }
    } catch (err) {
      if (sourceElement) showTooltip(sourceElement, 'Invalid path', true);
      return false;
    }

    if (editor.getMode() !== 'tree') {
      if (sourceElement) showTooltip(sourceElement, 'Switch to tree mode first', true);
      return false;
    }

    // Пытаемся найти узел и прокрутить
    try {
      const pathArray = pathStr.split('.').map(part => /^\d+$/.test(part) ? parseInt(part, 10) : part);
      let targetNode = null;
      if (editor.node && typeof editor.node.findNodeByPath === 'function') {
        targetNode = editor.node.findNodeByPath(pathArray);
      } else if (editor.tree && typeof editor.tree.findNodeByPath === 'function') {
        targetNode = editor.tree.findNodeByPath(pathArray);
      } else if (editor.editor && editor.editor.tree && typeof editor.editor.tree.findNodeByPath === 'function') {
        targetNode = editor.editor.tree.findNodeByPath(pathArray);
      }

      if (targetNode) {
        setTimeout(() => {
          if (typeof targetNode.scrollTo === 'function') {
            targetNode.scrollTo(() => { if (typeof targetNode.focus === 'function') targetNode.focus(); });
          } else if (targetNode.dom) {
            targetNode.dom.scrollIntoView({ behavior: 'smooth', block: 'center' });
          }
        }, 100);
        setTimeout(() => { lastSelectedPath = getCurrentPath(); }, 200);
        if (sourceElement) {
          const shortPath = pathStr.length > 50 ? pathStr.substring(0, 47) + '...' : pathStr;
          showTooltip(sourceElement, shortPath, false);
        }
        return true;
      }
    } catch (e) {}

    if (sourceElement) showTooltip(sourceElement, 'Cannot find node: ' + pathStr, true);
    return false;
  }

  function getSelectedTextFromAce() {
    const ace = editor.aceEditor;
    if (ace) {
      const selection = ace.getSession().getSelection();
      if (selection && !selection.isEmpty()) {
        return ace.getSession().getTextRange(selection.getRange());
      }
    }
    return '';
  }

  function getFullPathFromAceCursor() {
    try {
      const ace = editor.aceEditor;
      if (!ace) return '';
      const pos = ace.getCursorPosition();
      const session = ace.getSession();
      const doc = session.getDocument();
      const fullText = doc.getValue();
      let cursorPos = 0;
      for (let i = 0; i < pos.row; i++) cursorPos += doc.getLine(i).length + 1;
      cursorPos += pos.column;

      const textBefore = fullText.substring(0, cursorPos);
      const pathParts = [];
      let braceLevel = 0, inString = false, currentKey = '', searchingForKey = false;
      let arrayIndices = [], currentArrayIndex = -1;

      for (let i = 0; i < textBefore.length; i++) {
        const ch = textBefore[i];
        const prevCh = i > 0 ? textBefore[i-1] : '';
        if (ch === '"' && prevCh !== '\\') {
          inString = !inString;
          if (!inString && searchingForKey && currentKey) {
            pathParts.push(currentKey);
            currentKey = '';
            searchingForKey = false;
          }
          continue;
        }
        if (!inString) {
          if (ch === '{') braceLevel++;
          else if (ch === '}') {
            braceLevel = Math.max(0, braceLevel - 1);
            if (arrayIndices.length && braceLevel <= arrayIndices[arrayIndices.length-1].level) arrayIndices.pop();
          } else if (ch === '[') {
            braceLevel++;
            arrayIndices.push({ index: 0, level: braceLevel - 1 });
            currentArrayIndex = 0;
          } else if (ch === ']') {
            braceLevel = Math.max(0, braceLevel - 1);
            if (arrayIndices.length && braceLevel === arrayIndices[arrayIndices.length-1].level) arrayIndices.pop();
            currentArrayIndex = arrayIndices.length ? arrayIndices[arrayIndices.length-1].index : -1;
          } else if (ch === ',' && arrayIndices.length) {
            arrayIndices[arrayIndices.length-1].index++;
            currentArrayIndex = arrayIndices[arrayIndices.length-1].index;
          }
          if (ch === '"') { searchingForKey = true; currentKey = ''; }
          else if (searchingForKey && ch !== '"') currentKey += ch;
          else if (searchingForKey && ch === ':' && currentKey) {
            pathParts.push(currentKey);
            currentKey = '';
            searchingForKey = false;
          }
        }
      }
      if (arrayIndices.length && currentArrayIndex >= 0) pathParts.push(String(currentArrayIndex));
      if (pathParts.length) return pathParts.join('.');
      const simpleMatch = textBefore.match(/"([^"]+)"\s*:/g);
      if (simpleMatch && simpleMatch.length) {
        const lastMatch = simpleMatch[simpleMatch.length-1];
        const keyMatch = lastMatch.match(/"([^"]+)"/);
        if (keyMatch) return keyMatch[1];
      }
      return '';
    } catch(e) { return ''; }
  }

  function findPathByValue(obj, target, path = []) {
    if (obj === target) return path;
    if (obj && typeof obj === 'object') {
      if (Array.isArray(obj)) {
        for (let i = 0; i < obj.length; i++) {
          const found = findPathByValue(obj[i], target, [...path, i]);
          if (found) return found;
        }
      } else {
        for (const key in obj) {
          if (obj.hasOwnProperty(key)) {
            const found = findPathByValue(obj[key], target, [...path, key]);
            if (found) return found;
          }
        }
      }
    }
    return null;
  }

 function findPositionForPath(path, json) {
  if (!path) return -1;

  // 1. Нормализуем путь: "cells[1].name1" -> ["cells", "1", "name1"]
  const parts = path.replace(/\[(\d+)\]/g, '.$1').split('.').filter(Boolean);
  
  let searchPos = 0;
  let finalPos = -1;

  for (let i = 0; i < parts.length; i++) {
    const part = parts[i];
    const isIndex = /^\d+$/.test(part);

    if (isIndex) {
      // Это индекс массива. Ищем начало массива и пропускаем N элементов.
      const targetIndex = parseInt(part, 10);
      let arrayStart = json.indexOf('[', searchPos);
      if (arrayStart === -1) break;
      
      searchPos = arrayStart + 1;
      let count = 0;
      let bracketLevel = 0;
      let braceLevel = 0;
      let inString = false;

      // Проходим по символам, пока не найдем нужный индекс
      while (count < targetIndex && searchPos < json.length) {
        const char = json[searchPos];
        // Учитываем кавычки, чтобы не споткнуться о запятые внутри строк
        if (char === '"' && json[searchPos - 1] !== '\\') inString = !inString;
        if (!inString) {
          if (char === '[') bracketLevel++;
          else if (char === ']') bracketLevel--;
          else if (char === '{') braceLevel++;
          else if (char === '}') braceLevel--;
          else if (char === ',' && bracketLevel === 0 && braceLevel === 0) {
            count++;
          }
        }
        searchPos++;
      }
      finalPos = searchPos;
    } else {
      // Это ключ объекта. Ищем строку "ключ"
      const keyPattern = '"' + part + '"';
      const found = json.indexOf(keyPattern, searchPos);
      if (found !== -1) {
        finalPos = found;
        searchPos = found + keyPattern.length;
      } else {
        break; // Ключ не найден
      }
    }
  }

  return finalPos;
}

function setCursorAt(position) {
  const ace = editor.aceEditor;
  if (!ace) return;
  const doc = ace.getSession().getDocument();
  let total = 0;
  
  for (let i = 0; i < doc.getLength(); i++) {
    const len = doc.getLine(i).length + 1; // +1 для символа переноса строки
    if (total + len > position) {
      const row = i;
      const column = position - total;

      // Устанавливаем курсор
      ace.moveCursorTo(row, column);
      
      // Прокручиваем к строке:
      // row - номер строки
      // true - центрировать (строка будет посередине экрана)
      // true - анимировать прокрутку
      ace.scrollToLine(row, true, true, function () {});
      
      ace.focus();
      break;
    }
    total += len;
  }
}

  function onModeChange(newMode) {
    if (newMode === 'code') {
      if (lastSelectedPath) {
        setTimeout(() => {
          const text = editor.getText();
          const pos = findPositionForPath(lastSelectedPath, text);
          if (pos !== -1) setCursorAt(pos);
        }, 150);
      }
    } else if (newMode === 'tree') {
      setTimeout(() => {
        let path = null;
        const selected = getSelectedTextFromAce();
        if (selected) {
          try {
            const data = editor.get();
            let val;
            try { val = JSON.parse(selected); } catch(e) { val = selected; }
            const found = findPathByValue(data, val);
            if (found) path = found.join('.');
          } catch(e) {}
        }
        if (!path) path = getFullPathFromAceCursor();
        if (!path && lastSelectedPath) path = lastSelectedPath;
        if (path) selectNodeByPath(path, null);
      }, 200);
    }
  }

  // Статическое поле ввода пути
  function addStaticPathNav() {
    if (staticNavAdded) return true;
    const menu = document.querySelector('.jsoneditor-menu');
    if (!menu) return false;
    if (menu.querySelector('.static-path-nav')) return true;

    // Ищем элемент поиска (обычно .jsoneditor-search или его родитель)
    let searchElement = menu.querySelector('.jsoneditor-search');
      if (!searchElement) {
        var searchInput = menu.querySelector('input[placeholder*="search"]');
        if (searchInput) searchElement = searchInput.closest('.jsoneditor-search');
      }

    const navDiv = document.createElement('div');
    navDiv.className = 'static-path-nav';
    navDiv.innerHTML = `
      <input type="text" class="path-input" placeholder="Enter path (A.B)" autocomplete="off">
      <button class="path-go-btn">Go</button>
    `;

    const input = navDiv.querySelector('.path-input');
    const goBtn = navDiv.querySelector('.path-go-btn');

    if (lastSelectedPath) input.value = lastSelectedPath;

    function navigate() {
      const path = input.value.trim();
      if (path) {
        selectNodeByPath(path, goBtn);
        setTimeout(() => { lastSelectedPath = getCurrentPath(); }, 300);
      } else {
        showTooltip(goBtn, 'Enter a path', true);
        input.classList.add('pulse');
        setTimeout(() => input.classList.remove('pulse'), 500);
      }
    }

    goBtn.addEventListener('click', (e) => {
      e.stopPropagation();
      navigate();
    });
    input.addEventListener('keydown', (e) => {
      if (e.key === 'Enter') {
        e.preventDefault();
        e.stopPropagation();
        navigate();
      }
    });

    if (searchElement) {
      searchElement.parentNode.insertBefore(navDiv, searchElement);
    } else {
      menu.appendChild(navDiv);
    }

    staticNavAdded = true;
    return true;
  }

  // Кнопка копирования пути: теперь вставляется сразу после блока навигации (если он есть)
  function addCopyButton() {
    if (copyButtonAdded) return true;
    const menu = document.querySelector('.jsoneditor-menu');
    if (!menu) return false;
    if (menu.querySelector('.jsoneditor-copy-path')) return true;

    const btn = document.createElement('button');
    btn.className = 'jsoneditor-copy-path';
    btn.textContent = 'Copy';
    btn.title = 'Copy path';

    btn.addEventListener('click', async (e) => {
      e.stopPropagation();
      let path = '';
      if (editor.getMode() === 'tree') {
        path = getCurrentPath();
      } else {
        const selected = getSelectedTextFromAce();
        if (selected) {
          try {
            const data = editor.get();
            let val;
            try { val = JSON.parse(selected); } catch(e) { val = selected; }
            const found = findPathByValue(data, val);
            if (found) path = found.join('.');
          } catch(e) {}
        }
        if (!path) path = getFullPathFromAceCursor();
      }
      if (!path) {
        showTooltip(btn, 'No element selected', true);
        return;
      }
      try {
        await navigator.clipboard.writeText(path);
        const shortPath = path.length > 50 ? path.substring(0, 47) + '...' : path;
        showTooltip(btn, shortPath, false);
      } catch(err) {
        const ta = document.createElement('textarea');
        ta.value = path;
        document.body.appendChild(ta);
        ta.select();
        document.execCommand('copy');
        ta.remove();
        const shortPath = path.length > 50 ? path.substring(0, 47) + '...' : path;
        showTooltip(btn, shortPath, false);
      }
    });

    // Вставляем кнопку после статической навигации, если она есть, чтобы они были рядом
    const navDiv = menu.querySelector('.static-path-nav');
    if (navDiv && navDiv.nextSibling) {
      menu.insertBefore(btn, navDiv.nextSibling);
    } else if (navDiv) {
      menu.appendChild(btn);
    } else {
      // Если навигации ещё нет, вставляем в конец, но позже при повторном вызове будет скорректировано
      menu.appendChild(btn);
    }

    copyButtonAdded = true;
    return true;
  }

  function updatePath() {
    if (editor.getMode() === 'tree') {
      const p = getCurrentPath();
      if (p && p !== lastSelectedPath) {
        lastSelectedPath = p;
        const pathInput = document.querySelector('.static-path-nav .path-input');
        if (pathInput && pathInput.value !== p) pathInput.value = p;
      }
    }
  }

  // Инициализация редактора
  const options = {
    mode: 'code',
    modes: ['tree', 'code'],
    onModeChange: onModeChange,
    onChange: () => {
      if (document.getElementById('onChange')) document.getElementById('onChange').click();
    }
  };
  const editor = new JSONEditor(container, options);

  // Добавляем элементы управления после появления меню
  const obs = new MutationObserver(() => {
    addStaticPathNav();
    addCopyButton();
    const isTree = editor.getMode() === 'tree';
    if (isTree) {
      document.body.classList.remove('jsoneditor-mode-code');
      document.body.classList.add('jsoneditor-mode-tree');
    } else {
      document.body.classList.remove('jsoneditor-mode-tree');
      document.body.classList.add('jsoneditor-mode-code');
    }
  });
  obs.observe(document.body, { childList: true, subtree: true });

  setInterval(updatePath, 300);
  container.addEventListener('click', () => setTimeout(updatePath, 100));

  // Перехват кнопок навигации поиска (.jsoneditor-previous / .jsoneditor-next).
  // После каждого клика берём activeResult.node из searchBox редактора,
  // вызываем нативный _updateTreePath и синхронизируем поле ввода пути.
  function refreshSearchTreepath() {
    if (editor.getMode() !== 'tree') return;
    try {
      const sb = editor.searchBox;
      if (!sb || !sb.activeResult) return;
      const node = sb.activeResult.node;
      if (!node || typeof node.getNodePath !== 'function') return;
      editor._updateTreePath(node.getNodePath());
      const p = getCurrentPath();
      if (p) {
        lastSelectedPath = p;
        const pathInput = document.querySelector('.static-path-nav .path-input');
        if (pathInput && pathInput.value !== p) pathInput.value = p;
      }
    } catch(e) {}
  }

  document.addEventListener('click', (e) => {
    if (!e.target) return;
    if (!e.target.closest('.jsoneditor-previous, .jsoneditor-next')) return;
    setTimeout(refreshSearchTreepath, 0);
  }, true);

  setInterval(() => {
    if (!document.querySelector('.static-path-nav')) staticNavAdded = false;
    if (!document.querySelector('.jsoneditor-copy-path')) copyButtonAdded = false;
    addStaticPathNav();
    addCopyButton();
    const isTree = editor.getMode() === 'tree';
    if (isTree) {
      document.body.classList.remove('jsoneditor-mode-code');
      document.body.classList.add('jsoneditor-mode-tree');
    } else {
